Svelte API

Setting Data


component.set({
  questions: [
    'why is the sky blue?',
    'how do planes fly?',
    'where do babies come from?'
  ],
  answer: 'ask your mother'
});

Getting Data


component.get('answer');

// ask your mother

Watching Changes


const observer = component.observe('answer', answer => {
  console.log( `the answer is ${answer}` );
});

observer.cancel();

Respond to Events


const listener = component.on('thingHappened', event => {
  console.log(`A thing happened: ${event.thing}`);
});

listener.cancel();

Firing Events


component.fire('thingHappened', {
  thing: 'this event was fired'
});

Remove Component from DOM


component.destroy();

Template Syntax

Tags


<h1 style='color: {{color}};'>{{color}}</h1>
<p hidden='{{hideParagraph}}'>You can hide this paragraph.</p>

Triples


<p>Renders as: {{{html}}}</p>

If Blocks


{{#if user.loggedIn}}
  <a href='/logout'>log out</a>
{{else}}
  <a href='/login'>log in</a>
{{/if}}

Each Blocks


<div class='grid'>
  {{#each rows as row, y}}
    <div class='row'>
      {{#each columns as column, x}}
        <code class='cell'>
          {{x + 1}},{{y + 1}}:
          <strong>{{row[column]}}</strong>
        </code>
      {{/each}}
    </div>
  {{/each}}
</div>

Await Blocks


{{#await promise}}
  <p>wait for it...</p>
{{then answer}}
  <p>the answer is {{answer}}!</p>
{{catch error}}
  <p>well that's odd</p>
{{/await}}
